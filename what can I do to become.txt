public class TextComparison {
  private String input;
  private String mappedResult;
  private double comparisonThreshold;
  private List<Character> missingWord;
  private ResultSummary resultSummary;
  public Object originalSource;

  public TextComparison(
      String input,
      String mappedResult,
      double comparisonThreshold,
      List<Character> missingWord,
      ResultSummary resultSummary,
      Object originalSource) {
    this.input = input;
    this.mappedResult = mappedResult;
    this.comparisonThreshold = comparisonThreshold;
    this.missingWord = missingWord;
    this.resultSummary = resultSummary;
    this.originalSource = originalSource;
  }

  public String getInput() {
    return input;
  }

  public String getMappedResult() {
    return mappedResult;
  }

  public double getComparisonThreshold() {
    return comparisonThreshold;
  }

  public void setMappedResult(String mappedResult) {
    this.mappedResult = mappedResult;
  }

  public List<Character> getMissingWord() {
    return missingWord;
  }

  public ResultSummary getResultSummary() {
    return resultSummary;
  }

  @Override
  public String toString() {
    return "TextComparison{"
        + "input='"
        + input
        + '\''
        + ", mappedResult='"
        + mappedResult
        + '\''
        + ", comparisonThreshold="
        + comparisonThreshold
        + ", missingWord="
        + missingWord
        + ", resultSummary="
        + resultSummary
        + '}';
  }
}



  public static TextComparison checkMatch(String input, Set<String> data) {
    JaroWinklerSimilarity jaroWinkler = new JaroWinklerSimilarity();
    double threshold = 0.9;

    String mappedResult = null;
    double highestSimilarity = 0.0;
    List<Character> missingWord = new ArrayList<>();
    ResultSummary resultSummary = ResultSummary.NO_MATCH;
    Object originalSource = null;

    if (data.contains(input)) {
      mappedResult = input;
      highestSimilarity = 1.0;
      resultSummary = ResultSummary.EXACT_MATCH;
      originalSource = input;
    } else {
      for (String str : data) {
        double similarity = jaroWinkler.apply(input, str);
        if (similarity > highestSimilarity && similarity > threshold) {
          mappedResult = str;
          highestSimilarity = similarity;
          resultSummary = ResultSummary.CLOSE_MATCH;
          missingWord = new ArrayList<>();
          originalSource = str;

          // Find missing characters
          int i = 0, j = 0;
          while (i < input.length() && j < str.length()) {
            if (input.charAt(i) == str.charAt(j)) {
              i++;
              j++;
            } else {
              missingWord.add(str.charAt(j));
              j++;
            }
          }
          while (j < str.length()) {
            missingWord.add(str.charAt(j));
            j++;
          }
        }
      }
    }
    if (mappedResult == null) {
      mappedResult = input;
    }
    return new TextComparison(
        input, mappedResult, highestSimilarity, missingWord, resultSummary, originalSource);
  }

  public static TextComparison checkMatch(String input, Map<String,String> map){
    HashSet<String> hashSet = new HashSet<>();
    for(var m : map.entrySet()){
      hashSet.add(m.getKey());
    }
    return checkMatch(input, hashSet);
  }
  public static<T> TextComparison checkMatchWithList(String input, Map<String,List<T>> map){
    HashSet<String> hashSet = new HashSet<>();
    for(var m : map.entrySet()){
      hashSet.add(m.getKey());
    }
    return checkMatch(input, hashSet);
  }
